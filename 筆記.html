EJS

是一個簡易的模板語法

1.安裝

    
      npm i -D ejs
      npm i -D ejs-locals

2.引入 EJS 

    
      <!-- EJS 核心 -->
      const engine = require("ejs-locals");
      app.engine("ejs", engine);
      <!-- 讀取 EJS 檔案位置 -->
      app.set('views','./views');
      <!-- 用 EJS 引擎跑模板 -->
      app.set("view engine", "ejs");

      <!-- 引入 index.ejs -->
      app.get("/", (req, res) => {
        res.render("index");
      });
